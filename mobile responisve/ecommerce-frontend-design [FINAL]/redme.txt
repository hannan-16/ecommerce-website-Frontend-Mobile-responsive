        const categorySelect = document.getElementById('categorySelect');

            if (this.value === 'clothing') {
                navigateToProducts('clothing');
            } else if (this.value === 'electronics') {
                navigateToProducts('electronics');
            } else {
                navigateToProducts('all'); // Or however you want to handle "All Category"
            }
        });

        function navigateToProducts(category) {
            let url = 'product.html';
            if (category && category !== 'all') {
                url += `?category=${category}`;
            }
            window.location.href = url;
        }

        function updateAppliedFilters(category) {
            const appliedFiltersDiv = document.getElementById('appliedFilters');
            appliedFiltersDiv.innerHTML = '';
            if (category && category !== 'all') {
                const filterTag = document.createElement('span');
                filterTag.classList.add('filter-tag');
                filterTag.textContent = category.charAt(0).toUpperCase() + category.slice(1);
                const removeButton = document.createElement('button');
                removeButton.classList.add('remove-filter');
                removeButton.innerHTML = '<i class="fas fa-times"></i>';
                removeButton.addEventListener('click', () => {
                    const url = new URL(window.location.href);
                    url.searchParams.delete('category');
                    window.location.href = url.toString();
                });
                filterTag.appendChild(removeButton);
                appliedFiltersDiv.appendChild(filterTag);
            }
            const clearAll = document.createElement('a');
            clearAll.href = 'product.html';
            clearAll.classList.add('clear-filter');
            clearAll.id = 'clearFilter';
            clearAll.textContent = 'Clear all filter';
            appliedFiltersDiv.appendChild(clearAll);
        }

// Function to change the main product image
function changeMainImage(imageSrc) {
    document.querySelector('.main-image').src = imageSrc;
    // Optional: Update active thumbnail class
    document.querySelectorAll('.thumbnail').forEach(thumb => {
        thumb.classList.remove('active');
        if (thumb.querySelector('img').src === imageSrc) {
            thumb.classList.add('active');
        }
    });
}

// Function to handle tab switching
function openTab(evt, tabName) {
    var i, tabcontent, tablinks;
    tabcontent = document.getElementsByClassName("tabcontent");
    for (i = 0; i < tabcontent.length; i++) {
        tabcontent[i].style.display = "none";
    }
    tablinks = document.getElementsByClassName("tablinks");
    for (i = 0; i < tablinks.length; i++) {
        tablinks[i].className = tablinks[i].className.replace(" active", "");
    }
    document.getElementById(tabName).style.display = "block";
    evt.currentTarget.className += " active";
}

document.addEventListener('DOMContentLoaded', () => {
    const urlParams = new URLSearchParams(window.location.search);
    const productId = parseInt(urlParams.get('id'));
    const product = allProducts.find(p => p.id === productId);

    if (product) {
        document.title = `${product.title} - Ecommerce`;
        document.querySelector('.product-title').textContent = product.title;
        document.querySelector('.main-image').src = product.image;

        const thumbnailImagesDiv = document.querySelector('.thumbnail-images'); // Select the existing div
        thumbnailImagesDiv.innerHTML = ''; // Clear existing thumbnails

        // Add the main image as the first active thumbnail
        thumbnailImagesDiv.innerHTML += `<div class="thumbnail active" onclick="changeMainImage('${product.image}')"><img src="${product.image}" alt="${product.title}"></div>`;

        // Loop through product.thumbnails to create other thumbnails
        if (product.thumbnails && product.thumbnails.length > 0) {
            product.thumbnails.forEach(thumbUrl => {
                thumbnailImagesDiv.innerHTML += `
                    <div class="thumbnail" onclick="changeMainImage('${thumbUrl}')">
                        <img src="${thumbUrl}" alt="Thumbnail">
                    </div>
                `;
            });
        }
        // You might want to handle out of stock scenarios
        document.querySelector('.in-stock').innerHTML = `<i class="fas fa-check-circle"></i> In stock`;
        const ratingStars = Math.round(product.rating);
        let starsHTML = '';
        for (let i = 0; i < ratingStars; i++) {
            starsHTML += `<i class="fas fa-star" style="color: #ffc107;"></i>`;
        }
        for (let i = ratingStars; i < 5; i++) {
            starsHTML += `<i class="far fa-star" style="color: #ffc107;"></i>`;
        }
        document.querySelector('.product-rating').innerHTML = `${starsHTML} <span class="rating-count">(${product.orders} orders)</span>`;
        document.querySelector('.price-discounted').textContent = `$${product.price.toFixed(2)}`;
        if (product.oldPrice) {
            document.querySelector('.price-original').textContent = `$${product.oldPrice.toFixed(2)}`;
        } else {
            document.querySelector('.price-original').style.display = 'none';
        }

        const colorSelect = document.getElementById('color-select');
        const sizeSelect = document.getElementById('size-select');
        if (product.details && product.details.color) {
            colorSelect.innerHTML = product.details.color.map(color => `<option value="${color}">${color}</option>`).join('');
        } else {
            colorSelect.style.display = 'none';
            document.querySelector('label[for="color-select"]').style.display = 'none';
        }
        if (product.details && product.details.size) {
            sizeSelect.innerHTML = product.details.size.map(size => `<option value="${size}">${size}</option>`).join('');
        } else {
            sizeSelect.style.display = 'none';
            document.querySelector('label[for="size-select"]').style.display = 'none';
        }

        const productMetaDiv = document.querySelector('.product-info'); // Target the .product-info div
        // Clear existing product meta before adding new ones
